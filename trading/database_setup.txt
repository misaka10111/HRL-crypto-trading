create table trading_logs (
  id bigint generated by default as identity primary key,
  timestamp_utc text,
  macro_step integer,
  micro_step integer,
  btc_price numeric,
  total_portfolio_value numeric,
  cash_balance numeric,
  btc_holdings numeric,
  target_cash_pct numeric,
  target_btc_pct numeric,
  actual_cash_pct numeric,
  actual_btc_pct numeric,
  trade_action text
);
